{$CLEO .cs}
{$USE CLEO}

int FSToggle = 1
int result

while true    
    wait 0

    if and
        FSToggle == 1
        Pad.IsKeyPressed(KeyCode.F8)
    then
        while Pad.IsKeyPressed(KeyCode.F8)
            wait 0
        end
        result = Memory.Read(0x60A68E, 4, 1)
        Memory.Write(0x60A68E, 4, 0xFFF3615E, 1)
        FSToggle = 0
        print_help 'FS_OFF'
    end
    
    if and
        FSToggle == 0
        Pad.IsKeyPressed(KeyCode.F8)
    then
        while Pad.IsKeyPressed(KeyCode.F8)
            wait 0
        end
        Memory.Write(0x60A68E, 4, result, 1)
        FSToggle = 1
        print_help 'FS_ON'
    end
end

terminate_this_custom_script
